========= home.jsp ========= optional
set carousel rolling interval in ms
<div id="carousel-example-generic" class="carousel slide" data-ride="carousel" data-interval="10000">

set carousel content refresh interval in ms
setTimeout(function(){window.location.href="${pageContext.request.contextPath}";}, 1801000);

set carousel page numbers
List<Integer> idList = projectService.findTopIdList(5);



========= log4j.properties ========= !
set log location and size:
log4j.appender.dailyFile.File=D:/logs/logging.log4j
log4j.appender.dailyFile.MaxFileSize=1GB


========= applicationContext.xml ========= !
set custom parameters:
<bean id="settings" class="com.dataorta.projecthub.util.Settings">
	<property name="limit" value="12" />						<!-- items per page for project list and project detail -->
	<property name="fileServerPath" value="D:/Downloads/" />	<!-- this path is used for saving attachments -->
	<property name="recyclerPath" value="D:/myrecyclerpath/" />	<!-- this path is used for administrator to delete project, but not delete attachment files immediately -->
</bean>
